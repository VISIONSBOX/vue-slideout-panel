Setting panel width / height

custom width / height, can be a number like 10 or a string like '10px' 

const panel = this.$showPanel({
    component: "panel-1",
    width: {{ !isNaN(this.width) ? this.width : '"' + this.width + '"' }},
    openOn: "left",
    props: {}
});
const panel = this.$showPanel({
    component: "panel-1",
    height: {{ !isNaN(this.height) ? this.height : '"' + this.height + '"' }},
    openOn: "top",
    props: {}
});
